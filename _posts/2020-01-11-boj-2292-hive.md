import java.util.*;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		int num = sc.nextInt();
		int index = 1;
		int currentVal = 1;
		while(currentVal<num) {
			index++;
			currentVal += (index-1) * 6;
		}
		System.out.print(index);
	}
}
